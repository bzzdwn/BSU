	\subsubsection*{Постановка задачи.}
	Найти электрический и магнитный потенциал, электрическую и магнитную напряжённость внутри шара (внутри сферического слоя) при заданных диэлектрической проницаемости $\epsilon=-1$, условиях на электрический и магнитный потенциал ($u$ и $v$ соответственно) на его
	поверхности (на границах сферического слоя), и постоянной магнитной проницаемости,
	если распределение зарядов изменяется по закону $$\rho(r, \phi, \theta)=11r^6\cos ^4 \theta \sin^4 \theta (\cos 4\phi - \sin^2\phi).$$
	\begin{enumerate}
		\item $
		u|_{r=3}=4\cos ^2\theta \sin^4 \theta \sin^4 \phi.
		$
		\item $\begin{dcases}
			v|_{r=1} = 7\sin \theta \cos^3\theta \cos \phi,\\
			v|_{r=4}=6\sin^3\theta (\cos \phi  - \sin 3 \phi).
		\end{dcases}$
	\end{enumerate}
	\subsubsection*{Решение задачи.}
	Начнем c задачи для электрического потенциала. Считая
	$$\Delta u = - \dfrac{\rho(x,y,z)}{\epsilon},$$
	можем сформулировать задачу
	\begin{equation}
		\begin{cases}
			\Delta u = 11r^6\cos ^4 \theta \sin^4 \theta (\cos 4\phi - \sin^2\phi),\\
			u|_{r=3}=4\cos ^2\theta \sin^4 \theta \sin^4 \phi.
		\end{cases}
	\end{equation}
	Представим решение задачи в виде
	$$u(r,\theta,\phi)=w(r,\theta,\phi) + v(r,\theta,\phi),$$
	где 
	\begin{equation}
		\begin{cases}
			\Delta w = 0,\\
			w|_{r=3}=4\cos ^2\theta \sin^4 \theta \sin^4 \phi.
		\end{cases}
	\end{equation}
	\begin{equation}
		\begin{cases}
			\Delta v = 11r^6\cos ^4 \theta \sin^4 \theta (\cos 4\phi - \sin^2\phi),\\
			 v|_{r=3}=0,
		\end{cases}
	\end{equation}
	Начнем с отыскания решения задачи (2). Можно показать, что решение задачи (2) представимо в виде
	\begin{equation}
		w(r,\theta,\phi) = \sumnm (A_{nm} \cos m\phi + B_{nm}\sin m\phi) \cdot r^n \cdot P_n^{(m)}(\cos \theta),
	\end{equation}
	здесь мы учитываем неограниченность $r^{-(n+1)}$ при $r\to 0$. Подставим в выражение (4) граничное условие задачи (2):
	\begin{multline*}
		w(r,\theta,\phi)|_{r=3} = \sumnm (A_{nm} \cos m\phi + B_{nm}\sin m\phi) \cdot 3^n \cdot P_n^{(m)}(\cos \theta) = 4\cos ^2\theta \sin^4 \theta \sin^4 \phi.
	\end{multline*}
	В силу того, что
	\begin{multline*}
		\sin ^4 \phi = \left(\dfrac{1 - \cos 2 \phi}{2} \right)^2 = \dfrac{1 - 2\cos 2 \pi + \cos ^2 2 \phi}{4 } =  \dfrac{1 - 2\cos 2 \phi + \frac{1+\cos 4 \phi}2 }{4 }= \\ = \dfrac 3 8 \cos 0 \phi - \dfrac 12 \cos 2 \phi + \dfrac 18 \cos 4 \phi,
	\end{multline*}
	получаем
	\begin{multline*}
		w(r,\theta,\phi)|_{r=3} = \sumnm (A_{nm} \cos m\phi + B_{nm}\sin m\phi) \cdot 3^n \cdot P_n^{(m)}(\cos \theta) = \\ = \dfrac 3 2\cos ^2\theta \sin^4 \theta \cos 0 \phi  -2 \cos ^2\theta \sin^4 \theta \cos 2 \phi + \dfrac 1 2 \cos ^2\theta \sin^4 \theta \cos 4 \phi.
	\end{multline*}
	Таким образом, приравнивая соответствующие коэффициенты $\cos m\phi$, $\sin m \phi$, мы имеем
	$$A_{nm} = \begin{cases}
		A_{n0},\ m = 0,\\
		A_{n2},\ m = 2,\\
		A_{n4},\ m = 4,\\
		0, \ \text{иначе}.
	\end{cases}\ B_{nm} = 0,\ \forall m.$$
	Отсюда следует, что
	\begin{multline*}
		w(r,\theta,\phi)|_{r=3} = \sumn \Big(A_{n0}\cdot P_n^{(0)}(\cos \theta) + A_{n2}\cdot P_n^{(2)}(\cos \theta)\cos 2 \phi + A_{n4} \cdot P_n^{(4)}(\cos \theta)\cos 4 \phi\Big) \cdot 3^n  = \\ = \dfrac 3 2\cos ^2\theta \sin^4 \theta -2 \cos ^2\theta \sin^4 \theta \cos 2 \phi + \dfrac 1 2 \cos ^2\theta \sin^4 \theta \cos 4 \phi.
	\end{multline*}
	Приравняем соответствующие коэффициенты при $\cos m \phi$ и получим 3 разложения
	\begin{equation}
		\sumn (A_{n0}\cdot P_n^{(0)}(\cos \theta))\cdot 3^n = \dfrac 3 2\cos ^2\theta \sin^4 \theta,
	\end{equation}
	\begin{equation}
		\sumn (A_{n2}\cdot P_n^{(2)}(\cos \theta))\cdot 3 ^n = -2 \cos ^2\theta \sin^4 \theta,
	\end{equation}
	\begin{equation}
		\sumn (A_{n4}\cdot P_n^{(4)}(\cos \theta))\cdot 3 ^n = \dfrac 1 2 \cos ^2\theta \sin^4 \theta.
	\end{equation}
	Теперь будем вычислять коэффициенты разложений $A_{n0}$, $A_{n2}$, $A_{n4}$. Начнем с разложения (5). Определим степень правой части при замене $x = \cos \theta$:
